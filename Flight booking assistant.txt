        "import datetime\n",
        "import json\n",
        "import re\n",
        "import dateparser\n",
        "\n",
        "def fetch_flights(origin: str, destination: str, date: str, min_price=None, max_price=None, max_stops=None):\n",
        "    # Mock flight data\n",
        "    mock_flights = [\n",
        "        {\"flight_number\": \"AI203\", \"airline\": \"Air India\", \"price\": 4500, \"departure_time\": \"06:00\", \"arrival_time\": \"08:05\", \"stops\": 0},\n",
        "        {\"flight_number\": \"6E101\", \"airline\": \"IndiGo\", \"price\": 5100, \"departure_time\": \"07:30\", \"arrival_time\": \"08:45\", \"stops\": 0},\n",
        "        {\"flight_number\": \"UK304\", \"airline\": \"Vistara\", \"price\": 4800, \"departure_time\": \"09:00\", \"arrival_time\": \"11:30\", \"stops\": 1},\n",
        "        {\"flight_number\": \"AI450\", \"airline\": \"Air India\", \"price\": 7000, \"departure_time\": \"10:00\", \"arrival_time\": \"12:30\", \"stops\": 1},\n",
        "        {\"flight_number\": \"6E202\", \"airline\": \"IndiGo\", \"price\": 5500, \"departure_time\": \"11:00\", \"arrival_time\": \"13:30\", \"stops\": 0}\n",
        "    ]\n",
        "\n",
        "    # Apply price and stop filters if provided\n",
        "    filtered_flights = [\n",
        "        flight for flight in mock_flights\n",
        "        if (min_price is None or flight[\"price\"] >= min_price) and\n",
        "           (max_price is None or flight[\"price\"] <= max_price) and\n",
        "           (max_stops is None or flight[\"stops\"] <= max_stops)\n",
        "    ]\n",
        "\n",
        "    # Calculate duration for each filtered flight\n",
        "    for flight in filtered_flights:\n",
        "        dep = datetime.datetime.strptime(flight[\"departure_time\"], \"%H:%M\")\n",
        "        arr = datetime.datetime.strptime(flight[\"arrival_time\"], \"%H:%M\")\n",
        "        if arr < dep:\n",
        "            arr += datetime.timedelta(days=1)\n",
        "        flight[\"duration_minutes\"] = int((arr - dep).total_seconds() / 60)\n",
        "\n",
        "    # Find the cheapest and fastest flights\n",
        "    cheapest = min(filtered_flights, key=lambda x: x['price']) if filtered_flights else None\n",
        "    fastest = min(filtered_flights, key=lambda x: x['duration_minutes']) if filtered_flights else None\n",
        "\n",
        "    # Format price with \"Rs.\" instead of ₹\n",
        "    if cheapest:\n",
        "        cheapest[\"price\"] = f\"Rs. {cheapest['price']:,.0f}\"\n",
        "    if fastest:\n",
        "        fastest[\"price\"] = f\"Rs. {fastest['price']:,.0f}\"\n",
        "\n",
        "    return {\n",
        "        \"origin\": origin.title(),\n",
        "        \"destination\": destination.title(),\n",
        "        \"date\": date,\n",
        "        \"date_readable\": datetime.datetime.strptime(date, \"%Y-%m-%d\").strftime(\"%A, %B %d, %Y\"),\n",
        "        \"cheapest_flight\": cheapest,\n",
        "        \"fastest_flight\": fastest,\n",
        "        \"total_flights\": len(filtered_flights),\n",
        "        \"filters_applied\": {\n",
        "            \"min_price\": min_price,\n",
        "            \"max_price\": max_price,\n",
        "            \"max_stops\": max_stops\n",
        "        },\n",
        "        \"all_flights\": filtered_flights,\n",
        "        \"status\": \"Flights fetched successfully\"\n",
        "    }\n",
        "\n",
        "def parse_user_input(user_input: str):\n",
        "    # Extract origin and destination cities\n",
        "    match = re.search(r'from\\s+([\\w\\s]+?)\\s+to\\s+([\\w\\s]+?)(\\s+|$)', user_input, re.IGNORECASE)\n",
        "    if not match:\n",
        "        print(\"❌ Could not extract origin and destination. Please use the format: 'from <city> to <city>'\")\n",
        "        return None, None, None, None, None, None\n",
        "\n",
        "    origin = match.group(1).strip()\n",
        "    destination = match.group(2).strip()\n",
        "\n",
        "    if origin.lower() == destination.lower():\n",
        "        print(\"⚠️ Origin and destination cannot be the same.\")\n",
        "        return None, None, None, None, None, None\n",
        "\n",
        "    # Extract date from user input\n",
        "    date_match = re.search(r'(?:on|for)\\s+(.+)', user_input, re.IGNORECASE)\n",
        "    date_text = date_match.group(1).strip() if date_match else None\n",
        "\n",
        "    # If no date found, assume today and notify user\n",
        "    if not date_text:\n",
        "        print(\"⚠️ No date provided, assuming today's date.\")\n",
        "        parsed_date = datetime.date.today()\n",
        "    else:\n",
        "        parsed_date = dateparser.parse(date_text, settings={'PREFER_DATES_FROM': 'future'})\n",
        "        if not parsed_date:\n",
        "            print(\"⚠️ Could not parse date. Defaulting to today.\")\n",
        "            parsed_date = datetime.date.today()\n",
        "        elif parsed_date.date() < datetime.date.today():\n",
        "            print(\"⚠️ Travel date cannot be in the past.\")\n",
        "            return None, None, None, None, None, None\n",
        "\n",
        "    travel_date = parsed_date.strftime(\"%Y-%m-%d\")\n",
        "\n",
        "    # Extract price range and stop preferences\n",
        "    min_price, max_price, max_stops = None, None, None\n",
        "    price_match = re.search(r'price\\s+(\\d+)\\s+to\\s+(\\d+)', user_input, re.IGNORECASE)\n",
        "    if price_match:\n",
        "        min_price = int(price_match.group(1))\n",
        "        max_price = int(price_match.group(2))\n",
        "\n",
        "    stops_match = re.search(r'(\\d+)\\s+stops?', user_input, re.IGNORECASE)\n",
        "    if stops_match:\n",
        "        max_stops = int(stops_match.group(1))\n",
        "\n",
        "    return origin, destination, travel_date, min_price, max_price, max_stops\n",
        "\n",
        "def display_flight_summary(result):\n",
        "    print(f\"\\n📍 Route: {result['origin']} ➡️ {result['destination']} on {result['date_readable']}\")\n",
        "\n",
        "    # Display cheapest flight\n",
        "    if result['cheapest_flight']:\n",
        "        print(f\"✈️ Cheapest: {result['cheapest_flight']['flight_number']} ({result['cheapest_flight']['airline']}) - {result['cheapest_flight']['price']}\")\n",
        "    else:\n",
        "        print(\"✈️ No flights found matching the criteria.\")\n",
        "\n",
        "    # Display fastest flight\n",
        "    if result['fastest_flight']:\n",
        "        print(f\"⚡ Fastest: {result['fastest_flight']['flight_number']} - {result['fastest_flight']['duration_minutes']} mins\")\n",
        "    else:\n",
        "        print(\"⚡ No flights found matching the criteria.\")\n",
        "\n",
        "    # Display all flights\n",
        "    if not result['all_flights']:\n",
        "        print(\"⚠️ No flights found for your criteria.\")\n",
        "\n",
        "    print(\"\\n📦 AIgent JSON Response:\\n\")\n",
        "    print(json.dumps(result, indent=4))\n",
        "\n",
        "def run_agent_simulation():\n",
        "    print(\"👋 Welcome to FlightBookingAssistant (Simulated AIgent)\")\n",
        "    print(\"💬 Try queries like:\")\n",
        "    print(\"  ➤ Book a flight from Delhi to Mumbai on Friday\")\n",
        "    print(\"  ➤ I want to fly from Bangalore to Hyderabad for tomorrow\")\n",
        "    print(\"  ➤ Find flights from Delhi to Mumbai with price between 4500 to 5500 and 0 stops\\n\")\n",
        "\n",
        "    while True:\n",
        "        user_input = input(\"🗣️  Your travel query: \")\n",
        "\n",
        "        # Parse the user input and fetch flights\n",
        "        origin, destination, travel_date, min_price, max_price, max_stops = parse_user_input(user_input)\n",
        "\n",
        "        if origin and destination:\n",
        "            result = fetch_flights(origin, destination, travel_date, min_price, max_price, max_stops)\n",
        "            display_flight_summary(result)\n",
        "        else:\n",
        "            print(\"❌ Please use a valid query like: 'Book a flight from <city> to <city> on <date>'\")\n",
        "\n",
        "        cont = input(\"\\n🔁 Search another flight? (y/n): \").strip().lower()\n",
        "        if cont != 'y':\n",
        "            print(\"👋 Goodbye! Safe travels!\")\n",
        "            break\n",
        "\n",
        "if __name__ == \"__main__\":\n",
        "    run_agent_simulation()\n"
